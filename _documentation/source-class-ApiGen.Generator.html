 574:     /**
 575:      * Sorts and filters groups.
 576:      *
 577:      * @param array $groups
 578:      * @return array
 579:      */
 580:     private function sortGroups(array $groups)
 581:     {
 582:         // Don't generate only 'None' groups
 583:         if (1 === count($groups) && isset($groups['None'])) {
 584:             return array();
 585:         }
 586: 
 587:         $emptyList = array('classes' => array(), 'interfaces' => array(), 'traits' => array(), 'exceptions' => array(), 'constants' => array(), 'functions' => array());
 588: 
 589:         $groupNames = array_keys($groups);
 590:         $lowerGroupNames = array_flip(array_map(function($y) {
 591:             return strtolower($y);
 592:         }, $groupNames));
 593: 
 594:         foreach ($groupNames as $groupName) {
 595:             // Add missing parent groups
 596:             $parent = '';
 597:             foreach (explode('\\', $groupName) as $part) {
 598:                 $parent = ltrim($parent . '\\' . $part, '\\');
 599:                 if (!isset($lowerGroupNames[strtolower($parent)])) {
 600:                     $groups[$parent] = $emptyList;
 601:                 }
 602:             }
 603: 
 604:             // Add missing element types
 605:             foreach ($this->getElementTypes() as $type) {
 606:                 if (!isset($groups[$groupName][$type])) {
 607:                     $groups[$groupName][$type] = array();
 608:                 }
 609:             }
 610:         }
 611: 
 612:         $main = $this->config->main;
 613:         uksort($groups, function($one, $two) use ($main) {
 614:             // \ as separator has to be first
 615:             $one = str_replace('\\', ' ', $one);
 616:             $two = str_replace('\\', ' ', $two);
 617: 
 618:             if ($main) {
 619:                 if (0 === strpos($one, $main) && 0 !== strpos($two, $main)) {
 620:                     return -1;
 621:                 } elseif (0 !== strpos($one, $main) && 0 === strpos($two, $main)) {
 622:                     return 1;
 623:                 }
 624:             }
 625: 
 626:             return strcasecmp($one, $two);
 627:         });
 628: 
 629:         return $groups;
 630:     }
