166:         /**
167:      * Configuration function
168:      *
169:      * @param string $module Module name
170:      * @param mixed  $names The names of the configuration options to return or to store
171:      * @param mixed  $values The values to store
172:          * @param boolean $override Defaults to: true. If false, then the new value is not set if one already exists
173:      */
174:         public static function config($module, $names, $values=null, $override=true)
175:         {
176:                 // Initial values to work with
177:                 $return = null;
178: 
179:                 $config = isset(self::$config[$module]) ? self::$config[$module] : array();
180:                 $type   = array(
181:                         'names' => gettype($names),
182:                         'values'=> gettype($values)
183:                 );
184: 
185:                 // Return values
186:                 if ($type['values'] == 'NULL')
187:                 {
188:                         // One name
189:                         if ($type['names'] == 'string' && isset($config[ $names ]) )
190:                         {
191:                                 $return = $config[ $names ];
192:                         }
193:                         // More names
194:                         elseif ($type['names'] == 'array')
195:                         {
196:                                 foreach ($names as $key=>$value)
197:                                 {
198:                                         $return[ $value ] = isset($config[ $value ]) ? $config[ $value ] : null;
199:                                 }
200:                         }
201:                 }
202:                 // Set some values
203:                 else
204:                 {
205:                         // One value
206:                         if ($type['values'] == 'string')
207:                         {
208:                                 // One value, one name
209:                                 if ($type['names'] == 'string')
210:                                 {
211:                                         if ($override == false && isset(self::$config[$module][ $names ]))
212:                                         {
213:                                                 $return = self::$config[$module][ $names ];
214:                                         }
215:                                         else
216:                                         {
217:                                                 self::$config[$module][ $names ] = $values;
218:                                                 $return = $values;
219:                                         }
220:                                 }
221:                                 // One value, more names
222:                                 elseif ($type['names'] == 'array')
223:                                 {
224:                                         foreach ($names as $key=>$value)
225:                                         {
226:                                                 if ($override == false && isset(self::$config[$module][ $value ]))
227:                                                 {
228:                                                         $return = self::$config[$module][ $value ];
229:                                                 }
230:                                                 else
231:                                                 {
232:                                                         self::$config[$module][ $value ] = $values;
233:                                                         $return[ $value ] = $values;
234:                                                 }
235:                                         }
236:                                 }
237:                         }
238:                         // More values
239:                         elseif ($type['values'] == 'array')
240:                         {
241:                                 // Ensure that the $names is an array
242:                                 if ($type['names'] != 'array') { $names = array($names); }
243: 
244:                                 // Assign the values
245:                                 $names = array_values($names);
246:                                 $values = array_values($values);

248:                                 for ($a=0; $a++; $a<count($names))
249:                                 {

319:      * @param array $configuration The configuration for the created object. Parameters: path, class
320:      */
321:         public static function obj($name, $configuration=array())
322:         {       
323:                 // Check framework initialization
324:                 if (self::$initialized == false) { die('starfish::init() command must be run within your script!'); }
325: 
326:                 // Object exists
327:                 if (isset(self::$instances[ $name ]) && is_object( self::$instances[ $name ] ))
328:                 {
329:                         return self::$instances[ $name ];
330:                 }
331:                 else
332:                 {
333:                         // Check if a configuration already exists
334:                         if ( isset(self::$objects[$name]) && is_array(self::$objects[$name]) )
335:                         {
336:                                 $configuration = array_merge( self::$objects[$name], $configuration );
337:                         }
338: 
339:                         // Name of the class
340:                         $class = isset( self::$objects[$name]['class'] ) ? self::$objects[$name]['class'] : $name;
341: 
342:                         // Include the files, if needed
343:                         if (!class_exists($class))
344:                         {
345:                                 $path = isset( self::$objects[$name]['path'] ) ? self::$objects[$name]['path'] : null;
346: 
347:                                 if ($path != null)
348:                                 {
349:                                         require_once( $path );
350:                                 }
351:                                 else
352:                                 {
353:                                         $core       = self::config('_starfish', 'core_objects') . DIRECTORY_SEPARATOR . $name .'.php';
354:                                         $objects    = self::config('_starfish', 'root_objects') . DIRECTORY_SEPARATOR . $name .'.php';
355:                                         $application    = self::config('_starfish', 'app_objects') . DIRECTORY_SEPARATOR . $name .'.php';
356: 
357:                                         if (file_exists($core)) { require_once( $core ); }
358:                                         else if (file_exists($objects)) { require_once( $objects ); }
359:                                         else if (file_exists($application)) { require_once( $application ); }
360:                                 }
361:                         }
362: 
363:                         // Create the class
364:                         if (class_exists($class))
365:                         {
366:                                 // Create the object
367:                                 $object = new $class;
368: 
369:                                 // Run the init method, if it exists
370:                                 if (method_exists($object, 'init')) { $object->init(); }
371: 
372:                                 // Store the object
373:                                 self::$instances[$name] = $object;
374: 
375:                                 // Return the object
376:                                 return $object;
377:                         }
378:                 }
379: 
380:                 return null;
381:         }
